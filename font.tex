% 中文字体

% 中文默认字体：
% 均为衬线字体（roman serif）
% 我喜欢方正楷体，但方正楷体没有其他字重
% 可暂时用霞骛楷体作为粗体和斜体的替代
% 月星楷有多重字重，但是是古体字
\setCJKmainfont{FZKai-Z03}[BoldFont={LXGW WenKai Medium}, ItalicFont={LXGW WenKai Light}]

\setCJKsansfont{Source Han Sans SC}[BoldFont={Source Han Sans SC Medium}, ItalicFont={Source Han Sans SC Light}]
% 中文等宽字体：霞骛等宽楷体
\setCJKmonofont{LXGW WenKai Mono}[BoldFont={LXGW WenKai Mono Medium}, ItalicFont={LXGW WenKai Mono Light}]
